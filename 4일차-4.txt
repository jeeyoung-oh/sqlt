 -- OUTER 조인처리시  leading/ordered 힌트는 무시됨.
 -- outer조인도  NL.SM. HASH 방식모두 가능함.
  --use_nl, use_merge, use_hash
1>  select /*+use_nl( e d) leading(e d)*/ e.ename, d.dname
     from emp e, dept d
     where e.deptno(+)= d.deptno ; 

2> select /*+ use_nl (e d) leading(e d) */ e.ename, d.dname
    from emp e RIGHT outer JOIN  dept d ON ( e.deptno = d.deptno ) ; 
3> select /*+use_hash( e d) leading(e d) */ e.ename, d.dname
     from emp e, dept d
     where e.deptno(+)= d.deptno ; 
4> select /*+use_merge( e d) leading(e d)*/ e.ename, d.dname
     from emp e, dept d
     where e.deptno(+)= d.deptno ; 

--outer조인사용시   HASH 조인경우에만 먼저읽는  build테이블 변경가능함.
 -- SWAP_JOIN_INPUTS 힌트 사용해야함.  
5> select /*+use_hash( e d) swap_join_inputs(e)  */ e.ename, d.dname
     from emp e, dept d
     where e.deptno(+)= d.deptno ; 
---outer조인 사용시 주의점.
